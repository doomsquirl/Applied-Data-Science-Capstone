
-- TASK 1
SELECT DISTINCT 
	LAUNCH_SITE 
FROM 
	SPACEXTBL
;

-- TASK 2
SELECT 
	* 
FROM 
	SPACEXTBL
WHERE 
	LAUNCH_SITE LIKE 'CCA%'
LIMIT 5
;

-- TASK 3
SELECT 
	SUM(PAYLOAD_MASS__KG_) AS "TOTAL_PAYLOAD_MASS"
FROM 
	SPACEXTBL
WHERE 
	CUSTOMER = 'NASA (CRS)'
;

-- TASK 4
SELECT 
	AVG(PAYLOAD_MASS__KG_) AS "AVERAGE_PAYLOAD_MASS"
FROM 
	SPACEXTBL
WHERE 
	BOOSTER_VERSION = 'F9 v1.1'
;

-- TASK 5
SELECT 
	MIN(DATE) AS "MIN_DATE"
FROM 
	SPACEXTBL
WHERE 
	LANDING__OUTCOME = 'Success (ground pad)'
;

-- TASK 6
SELECT 
	BOOSTER_VERSION 
FROM 
	SPACEXTBL
WHERE
	LANDING__OUTCOME = 'Success (drone ship)'
	AND PAYLOAD_MASS__KG_ > 4000
	AND PAYLOAD_MASS__KG_ < 6000
;

-- TASK 7
SELECT 
	MISSION_OUTCOME, COUNT(*) AS "MISSION_COUNT"
FROM 
	SPACEXTBL
GROUP BY 
	MISSION_OUTCOME
ORDER BY
	 COUNT(*) DESC
	,MISSION_OUTCOME DESC
;

-- TASK 8
SELECT DISTINCT 
	BOOSTER_VERSION
FROM 
	SPACEXTBL
WHERE 
	PAYLOAD_MASS__KG_ = (SELECT MAX(PAYLOAD_MASS__KG_) FROM SPACEXTBL)
;

-- TASK 9
SELECT 
	LANDING__OUTCOME, BOOSTER_VERSION, LAUNCH_SITE
FROM 
	SPACEXTBL
WHERE 
	LANDING__OUTCOME = 'Failure (drone ship)'
	AND LEFT(DATE, 4) = '2015'
;

-- TASK 10
SELECT 
	LANDING__OUTCOME, COUNT(*) AS "MISSION_COUNT"
FROM 
	SPACEXTBL
WHERE 
	DATE BETWEEN '2010-06-04' AND '2017-03-20'
GROUP BY 
	LANDING__OUTCOME
ORDER BY 
	COUNT(*) DESC
;
